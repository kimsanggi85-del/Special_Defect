-- 특별불량 개선활동 보고서 테이블 CREATE TABLE IF NOT EXISTS special_defect_reports ( id BIGSERIAL PRIMARY KEY, doc_no TEXT, write_date DATE, dept TEXT, occur_date DATE, customer TEXT, part_no TEXT, lot_no TEXT, qty TEXT, process TEXT, grade TEXT, writer TEXT, claim_no TEXT, defect_desc TEXT, impact TEXT, before_img TEXT, before_desc TEXT, cause_occur TEXT, cause_outflow TEXT, cause_4m TEXT, emergency TEXT, emergency_date DATE, actions JSONB, verify_method TEXT, before_ppm TEXT, after_ppm TEXT, verify_result TEXT, standardize TEXT, horizontal TEXT, final_status TEXT, close_date DATE, qa_verifier TEXT, monitor_period TEXT, doc_revision TEXT, similar_check TEXT, remarks TEXT, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- 마스터 데이터 테이블 (발생부서/고객사/품번/발생공정) CREATE TABLE IF NOT EXISTS special_defect_master ( id BIGSERIAL PRIMARY KEY, category TEXT NOT NULL, -- dept, customer, partNo, process item_value TEXT NOT NULL, sort_order INT DEFAULT 0, created_at TIMESTAMPTZ DEFAULT NOW(), UNIQUE(category, item_value) ); -- RLS 활성화 (사내망 사용 시 정책 별도 설정) ALTER TABLE special_defect_reports ENABLE ROW LEVEL SECURITY; ALTER TABLE special_defect_master ENABLE ROW LEVEL SECURITY; -- 모든 사용자 읽기/쓰기 허용 정책 (사내용) CREATE POLICY "Enable all access for reports" ON special_defect_reports FOR ALL USING (true) WITH CHECK (true); CREATE POLICY "Enable all access for master" ON special_defect_master FOR ALL USING (true) WITH CHECK (true);
